## 🚀 快速开始

### 1. 引入样式文件

在HTML文件的 `<head>` 标签中添加:

```html
<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>页面标题</title>

    <!-- 引入统一的设计系统 -->
    <link rel="stylesheet" href="/css/main.css">
</head>
<body>
    <!-- 页面内容 -->
</body>
</html>
```

## 📢 Toast通知

### HTML结构

```html
<!-- Toast容器 (放在body底部) -->
<div class="toast-container" id="toast-container"></div>
```

### JavaScript控制

```javascript
// 显示Toast通知
function showToast(message, type = 'info') {
    const container = document.getElementById('toast-container');

    const toast = document.createElement('div');
    toast.className = `toast ${type}`;
    toast.innerHTML = `
        <div class="toast-icon">${getIcon(type)}</div>
        <div class="toast-content">
            <div class="toast-message">${message}</div>
        </div>
        <button class="toast-close" onclick="this.parentElement.remove()">✕</button>
    `;

    container.appendChild(toast);

    // 3秒后自动消失
    setTimeout(() => {
        toast.remove();
    }, 3000);
}

function getIcon(type) {
    const icons = {
        success: '✓',
        error: '✕',
        warning: '⚠',
        info: 'ℹ'
    };
    return icons[type] || icons.info;
}

// 使用示例
showToast('操作成功!', 'success');
showToast('发生错误!', 'error');
showToast('警告信息', 'warning');
showToast('提示信息', 'info');
```

    box-shadow: var(--shadow-md);

### JavaScript中使用

```javascript
// 获取CSS变量值
const primaryColor = getComputedStyle(document.documentElement)
    .getPropertyValue('--color-primary-500');

// 设置CSS变量
document.documentElement.style.setProperty('--color-primary-500', '#新颜色');
```

## 📱 响应式设计

### 断点系统

```
- xs: 0px (手机)
- sm: 640px (大手机/小平板)
- md: 768px (平板)
- lg: 1024px (小笔记本)
- xl: 1280px (桌面)
- 2xl: 1536px (大桌面)
```

### 响应式工具类使用

```html
<!-- 在不同屏幕尺寸显示不同列数 -->
<div class="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-4">
    <!-- 内容 -->
</div>

<!-- 响应式显示/隐藏 -->
<div class="hidden md:block">仅在平板及以上显示</div>
<div class="block md:hidden">仅在手机显示</div>

<!-- 响应式flex方向 -->
<div class="flex flex-col md:flex-row gap-4">
    <!-- 手机垂直,平板水平 -->
</div>
```
